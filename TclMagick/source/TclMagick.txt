############################################################################
Tcl interface to ImageMagick's Wand interface
     http://www.imagemagick.org
############################################################################
General design:

One main commands:
    magick - Create/delete Magick "wand","drawing","pixel" objects

Creating a Magick objects creates a command of the same name
    magickName subcmd ?args?

Deleting the Magick object also deletes the command.

############################################################################
TclMagick objects
############################################################################
magick create type ?name?           # Create a TclMagick object of type
    type="wand","drawing","pixel"
    returns: object name

magick delete name ?name? ...       # Delete TclMagick objects

magick names ?type?                 # Return all TclMagick objects (of type)

magick type name                    # Return the type of the object

magick types                        # Return valid type names
